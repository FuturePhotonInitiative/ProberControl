        '''
        Set the output switch to 1 -> ON or 0 -> OFF

        :param value: Specified state, defaults to 0 for OFF, 1 for ON
        :type value: Integer
        '''
        if value == 0:
            self.gpib.write(':OUTP OFF')
        if value == 1:
            self.gpib.write(':OUTP ON')
